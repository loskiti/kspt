#include <iostream>
# include <fstream>
using namespace std;
int main(){
	int m,n;
	ofstream f("C:\\Users\\Всемогущая\\Music\\4\\out.txt",ios::out);
	ifstream F;
	F.open("C:\\Users\\Всемогущая\\Music\\4\\in.txt", ios::in);
	F>>m>>n;
	int *k = new int[m,n];
	for (int i = 0; i <= n-1; i++) {
		for (int j=0; j<=m-1;j++){
			if ((i<n/2) && (j>=i) && (j<=m-i-1)){
				k[i,j]=i+1;
				f<<k[i,j];
			}
			else {if ((i>=n/2) && (j>=abs(n-i)) && (j<=m-abs(n-i))) {
				    k[i,j]=n-i;f<<k[i,j];
			     }
			      else {if (j<m/2){k[i,j]=j+1; f<<k[i,j];
				       }
			            else {k[i,j]=m-j;f<<k[i,j];
						};
				  };
			};
		}
		f<<endl;
	}
	F.close();
	f.close();
	delete []k;
	system("pause");
	return 0;
}
